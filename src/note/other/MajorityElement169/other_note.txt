    摩尔投票算法
public int majorityElementByMooreVoteAlgorithm(int[] nums) {
        int count = 0, ans = 0;
        for (int num : nums) {
            if (count == 0) {
                ans = num;
                count = 1;
            } else if (ans == num) {
                count++;
            } else {
                count--;
            }
        }
        return ans;
    }
