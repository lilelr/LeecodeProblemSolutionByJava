   public void invert(TreeNode root){
           if(root!=null){
               invert(root.left); //递归 invert
               invert(root.right); // 递归 invert
               if(root.left==null && root.right==null) return;
               TreeNode temp = root.left;
               root.left = root.right;
               root.right = temp;
           }
       }
